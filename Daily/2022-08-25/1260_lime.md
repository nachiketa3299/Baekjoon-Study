from collections import deque
n, m, s = map(int, input().split())
graphs = [[] for _ in range(n+1)]
for _ in range(m):
    v, e = map(int, input().split())
    graphs[v].append(e)
    graphs[e].append(v)

visited = [False]*(n+1)


def dfs(graph, s):
    if visited[s]:
        return
    print(s, end=' ')
    visited[s] = True
    for n in sorted(graphs[s]):
        if not visited[n]:
            dfs(graph, n)
    return
